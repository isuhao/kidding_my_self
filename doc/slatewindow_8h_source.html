00127         void updateNodeId(uint oldId, uint newId) {
00128                 int pos = nodes.indexOf(oldId);
00129                 if (pos >= 0) {
00130                         nodes.removeAt(pos);
00131                         nodes.append(newId);
00132                 }
00133                 for (QList<LinkInfo>::iterator it = visibleNodeLinks.begin(); it != visibleNodeLinks.end(); it++) { // uso un iterador para poder modificar
00134                         if ((*it).getNode() == oldId)
00135                                 (*it).setNode(newId);
00136                 }
00137         }

00140                 int pos = subgroups.indexOf(oldId);

00142                         subgroups.removeAt(pos);

00144                 }
00145                 for (QList<LinkInfo>::iterator it = visibleSubgroupLinks.begin(); it != visibleSubgroupLinks.end(); it++) { // uso un iterador para poder modificar
00146                         if ((*it).getNode() == oldId)
00147                                 (*it).setNode(newId);
00148                 }
00149         }
