00068 /* Sparse matrix representation using Compressed Row Storage (CRS) format.
00069  * See http://www.netlib.org/linalg/html_templates/node91.html#SECTION00931100000000000000
00070  */
00071 
00072 struct sba_crsm{
00073     int nr, nc;   /* #rows, #cols for the sparse matrix */
00074     int nnz;      /* number of nonzero array elements */
00075     int *val;     /* storage for nonzero array elements. size: nnz */
00076     int *colidx;  /* column indexes of nonzero elements. size: nnz */
00077     int *rowptr;  /* locations in val that start a row. size: nr+1.
00078                    * By convention, rowptr[nr]=nnz
00079                    */
00080 };
