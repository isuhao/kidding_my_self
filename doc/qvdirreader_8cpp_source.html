00053         } else { // Will read only selected files (i.e. using wildcards) in specified dir:
00054             QString path = info.path(), files = info.fileName();
00055             QFileInfo info(path);
00056             if(not info.isDir())
00057                 return false;
00058             dir = QDir(path);
00059             filters << files;

00063         filenames = dir.entryList();
00064 
00065         if(filenames.isEmpty())
00066             return false;

00070         realTime = (suggested_options & QVVideoReader::RealTime);
00071 
00072         if(realTime)
00073             suggested_fps = 25;
00074 
00075         // std::cout << qPrintable(dir.absolutePath()) << "\n";
00076 
00077         // Will not rescale, answer size (0,0)
00078         suggested_cols = suggested_rows = 0;

00080         if(filenames.size() > 0) {
00081             index = 0;
00082             return true;
